--a a)	Para la categoría ‘Prenda’ muestra los montos y unidades vendidas en ‘Europe' y ‘Pacific’
-- (grupos territoriales) en el segundo trimestre del año antepasado.
select
{ [Measures].[Sales Amount], [Measures].[Order Quantity] } on columns,
filter( {[Sales Territory].[Sales Territory Group].&[Europe], [Sales Territory].[Sales Territory Group].&[Pacific]}, [Order Date].[Tiempo].[Año].&[2003].&[2] )on rows
from [Cubo]
where [Product].[Spanish Product Category Name].Prenda


--c)	Obtén los precios unitarios de cada línea de producto por grupos territoriales 
--de los empleados en el primero y tercer trimestres del año ante-antepasado. 
--En el resultado deben verse los grupos y los trimestres.
select 
{ [Employee - Sales Territory].[Sales Territory Group].members } on columns, 
{ [Order Date].[Tiempo].[Año].&[2002].&[1], [Order Date].[Tiempo].[Año].&[2002].&[3] } on rows
from [Cubo]
where [Measures].[Unit Price]

--d)	Muestra los montos de impuestos por país y por línea de productos, 
--considerando sólo a los que superaron los $20,000.
select 
{ [Reseller].[Jerarquía].[Sales Territory Country].members } on columns,
 filter( [Product].[Produc Line Name].members, [Measures].[Tax Amt] > 20000 )  on rows
from [Cubo]
where  [Measures].[Tax Amt]

-- l.
select non empty ([Reseller].[Jerarquía].[Sales Territory Country].members,
Product.[Product Line].members) on 0,
  	non empty (Product.[Model name].members, [Order Date].[2018]) on 1
from [Cubo]
where [Sales Amount]

--e)	Lista los costos totales y los montos vendidos en los trimestres 2 y 4, de los años antepasado y pasado, por cada categoría de los productos y por nombre de los empleados. Debe verse cada trimestre.

select
{ [Measures].[Total Product Cost], [Measures].[Sales Amount] } on columns, 
non empty ([Employee].[Employee].[FullName].members , [Product].[Spanish Product Category Name].members, { [Order Date].[Tiempo].[Año].&[2002].&[2], [Order Date].[Tiempo].[Año].&[2002].&[4], [Order Date].[Tiempo].[Año].&[2003].&[4], [Order Date].[Tiempo].[Año].&[2003].&[2] } ) on rows

--a)	Lista el nombre de los proyectos controlados por un departamento con clave máxima de 4,
-- cuyo nombre, del proyecto, inicie con una letra inferior a ‘O’.

select Doc.query('companyDB//project[@controllingDepartment<=4]/pname[.<"O"]')
from Company;

--b)	Para los empleados que viven en Texas (TX), escribe el nombre de sus hijos ('Son' o 'Daughter').

select Doc.query('companyDB//employee[contains(address, "Houston")]//dependent[relationship = "Son" or relationship = "Daughter"]/dependentName')
from Company;

--XQuery

--c)	Para los departamentos cuyo nombre inicia con una letra inferior a ‘P’,
-- muestra su nombre y la cantidad de ubicaciones (locations) que tienen.

select Doc.query
('for $x in //department
where $x/dname < "P"
return <resul> {$x/dname cast as xs:string?, count($x//location) cast as xs:string? } </resul>')
from Company;

--d)	Para los empleados que trabajan para el departamento 5, lista su nombre completo 
--junto con el nombre de sus dependientes, considerando sólo a los que tienen dependientes.

select Doc.query
('for $x in //employee, $y in $x//dependent
where $x/@worksFor = 5 and count($x//dependent) > 0 
return <resul> {$x/lname, $y/dependentName} </resul>')
from Company;

--e)	Para los departamentos que tienen más de dos ubicaciones, 
--escribir el nombre del departamento y el nombre (fname) de los empleados adscritos a él.

select Doc.query
('for $x in //department, $y in //employee
where $y/@worksFor = $x/@dno and count($x//location) > 2 
return <resul> {$x/dname, $y/fname} </resul>')
from Company;
